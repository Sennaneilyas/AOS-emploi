Add to User model — profile completion calculator

public function getProfileCompletionPercentAttribute(): int {
  $fields = ['name','phone','bio','avatar_url','city','occupation','budget','languages','social_link'];
  $filled = collect($fields)->filter(fn($f) => !empty($this->$f))->count();
  $answersDone = $this->questionnaireAnswers()->count() > 0 ? 1 : 0;
  return (int)(($filled + $answersDone) / (count($fields) + 1) * 100);
}

$answered = $request->user()->questionnaireAnswers()->count();

// Invalidate compatibility score cache for this user
CompatibilityScore::where('user_id_1', $request->user()->id)
  ->orWhere('user_id_2', $request->user()->id)

]);

{ "saved": true, "completion_percent": 62 }
The cache invalidation line is critical. Every time a user updates an answer, their existing compatibility scores with all other users become stale. Delete them here so they get recalculated on next search.
